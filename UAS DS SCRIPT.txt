    create table `EmailsData` (
        `a_id` INT not null auto_increment,
        `ppId` INT,
        `emailFrom` VARCHAR(50),
        `emailTo` VARCHAR(50),
        `emailSubject` VARCHAR(255),
        `emailBody` VARCHAR(4000),
        `sentOn` DATETIME,
        primary key (`a_id`)
    );
    create table `PendingAssetReportData` (
        `Assetidentifiers` VARCHAR(200) not null,
        `AssetType` VARCHAR(200),
        `assetDescription` VARCHAR(500),
        `accountnumber` VARCHAR(255),
        `assettypecp` VARCHAR(255),
        `accountname` VARCHAR(255),
        `processStatus` VARCHAR(255),

    );

    create table `UAC_Asset_Information` (
        `AssetId` INT not null auto_increment,
        `Assetidentifiers` VARCHAR(50) not null unique,
